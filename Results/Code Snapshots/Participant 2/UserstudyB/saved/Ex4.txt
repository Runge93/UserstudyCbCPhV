
public class FirstExercise {
	
	/*@
	@ normal_behavior
	@ requires a.length > 0;
	@ ensures (\forall int q; q >= 0 & q < a.length; a[\result]>=a[q]);
	@*/ 
	public static int maxElement(int[] a) {
		int maxIndex = 0;
		int maxValue = Integer.MIN_VALUE;
		/*@ loop_invariant i < a.length;
		  @ decreases i;
		  @*/
		for(int i = a.length - 1; i >= 0; i--) {
			if(maxValue < a[i]) {
				maxValue = a[i];
				maxIndex = i;
			}
		}
		return maxIndex;
	}
}
